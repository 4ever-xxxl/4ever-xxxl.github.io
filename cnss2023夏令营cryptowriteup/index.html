        tmp  = state & mask & lenmask
        output = 0
        while tmp != 0:
            output ^^= (tmp & 1)

        state = nextstate

def dfs(p, q, x):    
    if x == SZ:
        if p * q == n:
            global P,Q
            P = p
            Q = q
        return
    lenmask = 2**(x+1) - 1
    _p = [0,1] if p_bits[x] == None else [p_bits[x]]
    _q = [0,1] if q_bits[x] == None else [q_bits[x]]
    for pi in _p:
        for qi in _q:
            next_p = pi<<x | p
            next_q = qi<<x | q
            if next_p * next_q & lenmask == n & lenmask:
                dfs(next_p, next_q, x + 1)

dfs(1, 1, 1)
phi = (P - 1) * (Q - 1)
d = inverse_mod(e, phi)
flag = pow(c, d, n)
